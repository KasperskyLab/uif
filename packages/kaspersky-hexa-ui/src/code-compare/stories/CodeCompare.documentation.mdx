function App() {

    alert('left: ' + (left?.value ?? '—') + ', right: ' + (right?.value ?? '—'))
    setOldVersion(left)

    setOldValue(left ? codeCompareStoryCodeByVersion[left.value] : undefined)

      newValue={newValue}

}`}
